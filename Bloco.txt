var gerarNonce = function (length) {
    var text = "";
    var possible = "QWERTYUIOPASDFGHJKLZXCVBNMqwertyuiopasdfghjklzxcvbnm1234567890";
    for (var i = 0; i < length; i++) {
        text += possible.charCodeAt(Math.floor(Math.random() * possible.length));
    }
    return text;
}
var nonce = gerarNonce(15);
console.log(nonce);









// Selecionar a tag meta CSP com querySelector
const metaElement = document.querySelector('meta[http-equiv="Content-Security-Policy"]');
// Atualizar o valor do content, inserindo o nonce
metaElement.setAttribute('content', `script-src 'self' 'nonce-${nonce}';`);

console.log('Meta tag CSP atualizada: ', metaElement.getAttribute('content'));

var meta = document.getElementsByTagName("meta")[2];
var node_meta = document.importNode(meta, true);
console.log(node_meta);

metaElement.addEventListener('DOMContentLoaded', function () {
    // Atualizar o valor do content, inserindo o nonce
    metaElement.setAttribute('content', `script-src 'self' 'nonce-${nonce}';`);

    console.log('Meta tag CSP atualizada: ', metaElement.getAttribute('content'));

    var meta = document.getElementsByTagName("meta")[2];
    var node_meta = document.importNode(meta, true);
    console.log(node_meta);
});
